学院管理员表

CREATE TABLE  "SYS_ADMIN" 
   (	"MNO" CHAR(10) NOT NULL ENABLE, 
	"MNAME" CHAR(15) NOT NULL ENABLE, 
	"MPASSWORD" CHAR(30) NOT NULL ENABLE, 
	"MTEL" CHAR(20), 
	"MSEX" CHAR(3), 
	 PRIMARY KEY ("MNO") ENABLE
   ) ;


邮件信息表

CREATE TABLE  "SYS_MAIL" 
   (	"SENDER" CHAR(10) NOT NULL ENABLE, 
	"RECEIVER" CHAR(10) NOT NULL ENABLE, 
	"CONTENT" CHAR(420) NOT NULL ENABLE, 
	"SENDER_NAME" CHAR(51) NOT NULL ENABLE, 
	"IFREAD" CHAR(1) NOT NULL ENABLE, 
	"MNO" CHAR(4) NOT NULL ENABLE, 
	 CONSTRAINT "PK_SYS_MAIL" PRIMARY KEY ("MNO") ENABLE
   ) ;
